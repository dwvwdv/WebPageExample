request

  • urlopen(url , data = None ,[timeout ]*)
  • Request(url , data = None ,header = {}, method = None)

urlopen默認GET request,傳參後為 POST request

urlopen 參數 timeout 設置請求超時時間

requests庫

get(url , cookies , params , headers , timout)

  • cookies 
url = 'https://httpbin.org/cookies'
cookies = dict(cookies_name = "cookies_value")
response = requests.get(url , cookies = cookies)
response.text

#result -> '{"cookies": {"cookies_name": "cookies_value"}}'
  • params
url = 'https://httpbin.org/get'
payload = {"key" : ["value","666"]}
response = requests.get(url ,params = payload)
print(response.text)

"""
"args": {
    "key": [
      "value", 
      "666"
    ]
  },...
"""
  • headers
url = 'https://httpbin.org/get'
headers = {'user-agent': 'dwvwdv/0.0.1'}
response = requests.get(url ,headers = headers)
print(response.text)

"User-Agent": "dwvwdv/0.0.1"

post(url , data , files , json)

  • data
url = 'https://httpbin.org/post'
payload = {"key" : "value"}
response = requests.post(url , payload)
response.text


"""
"args": {}, 
  "data": "", 
  "files": {}, 
  "form": {
    "key": "value"  #payload data
  }...
"""
  • files
url = 'https://httpbin.org/post'
files = {'file': open('upload.txt','rb')}
response = requests.post(url ,files = files)
print(response.text)

"""
"files": {
    "file": "# -*- coding: utf-8 -*-\r\n\"\"\"\r\nCreated on Sat Mar 27 17:26:38 2021\r\n\r\n@author: USER\r\n\"\"\"\r\nimport requests\r\n\r\nurl = 'https://httpbin.org/post'\r\nfiles = {'file': open('nb.txt','rb')}\r\nresponse = requests.post(url ,files = files)\r\nprint(response.text)"
  },
"""
  • json
url = 'https://httpbin.org/post'
payload = {"key" : ["value",'666']}
response = requests.post(url ,json = payload)
print(response.text)

"""
"data": "{\"key\": [\"value\", \"666\"]}",
...中間略
"json": {
    "key": [
      "value", 
      "666"
    ]
  },
"""
