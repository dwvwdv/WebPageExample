控制角色移動

  • Rigidbody移動
  • CharacterController移動
  • UpdateLocation更新座標

三種移動方式實現，Rigidbody和CharacterController移動更適合3D，CharacterController去除了rigidbody的一些特質，只保留了部分(如重力，此外CharacterController的simpleMove非常難用，在跳躍檢測表現非常糟糕，一般不使用，UpdateLocation如果遇到碰撞沒辦法自動檢測，較不理想，但UpdateLocation很適合沒有障礙物、無視障礙物下的平面座標移動。

public class PlayerControl : MonoBehaviour
{
    Vector3 newPosition;
    public float moveSpeed = 5;

		//Rigidbody移動
    Rigidbody rig;
    CapsuleCollider col;

		//CharacterController移動
    CharacterController CCtrl;
    public bool isSimple;

    public enum MoveType
    {
        UpdateLocation,
        RigidbodyMove,
        CCtrlMove
    }

    public MoveType moveType;
    
    void Start()
    {
        switch (moveType)
        {
            case MoveType.RigidbodyMove:           
                rig = GetComponent<Rigidbody>();
                col = GetComponent<CapsuleCollider>();
                if (col == null)
                    col = this.gameObject.AddComponent<CapsuleCollider>();
                if (rig == null)
                    rig = this.gameObject.AddComponent<Rigidbody>();
                rig.constraints = RigidbodyConstraints.FreezeRotation;
                col.height = 1.5f;
                col.center = new Vector3(0, 0.66f, 0);
                break;
            case MoveType.CCtrlMove:
                CCtrl = GetComponent<CharacterController>();
                if (CCtrl == null)
                    CCtrl = this.gameObject.AddComponent<CharacterController>();
                CCtrl.height = 1.7f;
                CCtrl.center = new Vector3(0, 0.66f, 0);
                break;
        }
        
    }

    void Update()
    {
        computeMove();
        move();
    }

		//計算移動點數
    void computeMove()
    {
        Vector3 moveDir = new Vector3(Input.GetAxis("Horizontal"), 0, Input.GetAxis("Vertical"));
        newPosition = moveDir * moveSpeed * Time.deltaTime;
    }

		//選擇方式移動
    private void move()
    {
        switch (moveType)
        {
            case MoveType.RigidbodyMove:
                int velSpeed = 50;
                rig.velocity = newPosition * velSpeed;
                break;
            case MoveType.CCtrlMove:
                if (isSimple)
                    CCtrl.SimpleMove(newPosition);
                else
                    CCtrl.Move(newPosition);
                break;
            case MoveType.UpdateLocation:
                this.transform.position += newPosition;
                break;
        }
    }
}
